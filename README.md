    font-family: 'Segoe UI', sans-serif;
    margin: 0;
    padding: 20px;
  }

  h2, h3 {
    color: #FFD700; /* Gelb für Titel */
  }

  table {
    width: 100%;
    border-collapse: collapse;
    margin-top: 20px;
    background-color: #3b0000; /* Dunkelrot für Tabellenhintergrund */
    color: #FFD700; /* Gelbe Schrift */
  }

  th, td {
    border: 1px solid #990000;
    padding: 10px;
    text-align: center;
  }

  th {
    background-color: #660000; /* Dunkelrot für Tabellenkopf */
  }

  tr:nth-child(even) {
    background-color: #4b0000; /* Noch dunkleres Rot für abwechselnde Zeilen */
  }

  a {
    color: #FFD700; /* Gelb für Links */
    text-decoration: underline;
  }

  blockquote {
    font-style: italic;
    color: #FFD700; /* Gelb für Zitattext */
    border-left: 4px solid #990000;
    background-color: #3b0000; /* Dunkelrot für Zitatbox */
    padding-left: 10px;
  }

  img {
    display: block;
    margin: 0 auto 20px auto;
    max-width: 100%;
    height: auto;
  }
</style>
